SELECT SUM(SALARY) as summ FROM S_HREMPLOYEES sh ;
SELECT MIN(SALARY) as mini, MAX(SALARY) as maxi, AVG(SALARY) as avrg FROM S_HREMPLOYEES sh ;

SELECT COUNT(*),JOB_ID FROM S_HREMPLOYEES sh
GROUP BY JOB_ID ;

SELECT COUNT(1),JOB_ID, DEPARTMENT_ID from DBO.S_HREMPLOYEES by DEPARTMENT_ID having count(1)>10;

SELECT SUM(SALARY),JOB_ID
FROM DBO.S_HREMPLOYEES sh
WHERE JOB_ID IS NOT NULL
GROUP BY JOB_ID
HAVING SUM(SALARY)>100000
ORDER BY JOB_ID DESC;
